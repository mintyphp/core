<?php

use MintyPHP\Debugger;
use MintyPHP\Router;
use MintyPHP\DB;

use MintyPHP\Tools\Adminer;
use MintyPHP\Network;

// Change directory to project root
chdir(__DIR__ . '/..');
// Load the libraries
require_once 'vendor/autoload.php';
// Load the config parameters
require_once 'config/config.php';

if (Debugger::$enabled) {
    Debugger::$enabled = false;
    if (Network::isLocalIP($_SERVER['REMOTE_ADDR'] ?? '')) {
        die((new Adminer(DB::$host, DB::$port, DB::$username, DB::$password, DB::$database))->run());
    }
}
Router::redirect('/');
